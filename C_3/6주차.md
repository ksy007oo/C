### while문 
- 주어진 조건이 만족되는 동안 문장들을 반복 실행한다.
------------------------------------------------------
ex)
```c
while (i < 10 ) {
  printf("Hello World!\n");
  i++
  }
  ```
 ------------------------------------------------------
 ### while문 예제
 ```c
 #define _CRT_SECURE_NO_WARNINGS
 #include <stdio.h>
 
 int main(void)
 {
    int i = 0;
    
    while ( i < 5 )
    {
        printf("Hello World!");
        i++
     }
     
     return 0;
  }
 ```
 ------------------------------------
 예제 1
 ```c
 #define _CRT_SECURE_NO_WARNINGS
 #include <stdio.h>
 
 int main(void)
 {
    int x, y=1;
    printf("출력하고 싶은 단을 입력하시오:");
    scanf("%d",&x);
    
    while ( y <= 9 )
    {
        printf("%d*%d=%d\n", x , y++ , x * y);
    }
    
    return 0;
}
